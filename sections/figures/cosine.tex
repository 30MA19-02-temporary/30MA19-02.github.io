% !TeX root = ../../main.tex
\documentclass[tikz]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=newest}
\begin{document}
\begin{tikzpicture}[
        declare function={
                cosine(\x,\y)= (\x>=0) * cos(\x*deg(\y)) + (\x<0) * cosh(\x*\y);
            },
    ]
    \begin{axis}[
            colormap/hot,
            xlabel = \(k\),
            ylabel = \(x\),
            zlabel = \(\cos_k\left(x\right)\),
            domain=-2:2,
            domain y = -pi:+pi,
            ytick = {-pi,-pi/2,0,+pi/2,+pi},
            zmin = -3,
            zmax = +3,
            point meta min = -3,
            point meta max = +3,
            view = {150}{60},
            samples = 50,
        ]
        \addplot3 [
            surf,
            shader = faceted interp,
        ] {cosine(\x,\y)};
        \addplot3 [
            black,
            smooth,
        ](0, \y, 1);
        \addplot3 [
            black,
            smooth,
        ](1, \y, {cos(deg(\y))});
        \addplot3 [
            black,
            smooth,
        ](-1, \y, {cosh(-\y)});
    \end{axis}
\end{tikzpicture}
\end{document}
